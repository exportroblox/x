-- Valerie Prompt Script
-- Sound Configuration
local SOUND_CONFIG = {
	Mouth = {
		Volume = 10,
		IDs = {
			"rbxassetid://114489279570769",
			"rbxassetid://138966281638634",
			"rbxassetid://132420706005896"
		}
	},
	Gag = {
		Volume = 1,
		IDs = {
			"rbxassetid://117745817739081",
			"rbxassetid://123721935495656",
			"rbxassetid://119019430569155"
		}
	},
	Throat = {
		Volume = 1,
		IDs = {
			"rbxassetid://125819546977754",
			"rbxassetid://114686177700785",
			"rbxassetid://82767386277671"
		}
	}
}

-- Services
local ChatService = game:GetService("Chat")
local Debris = game:GetService("Debris")

-- Variables
local Debounce = false
local WhiteFade = false
local player

-- Helper Functions
local function PlayRandomSound(soundType, parent)
	local config = SOUND_CONFIG[soundType]
	if not config then return end

	local sound = Instance.new("Sound")
	sound.Name = soundType .. "Sound"
	sound.Parent = parent
	sound.Volume = config.Volume
	sound.SoundId = config.IDs[math.random(1, #config.IDs)]
	sound:Play()
	Debris:AddItem(sound, 1)
end

local function SetupCharacter(char)
	script.Parent.Parent.HumanoidRootPart.Anchored = true
	char.HumanoidRootPart.Anchored = true
	char.HumanoidRootPart.CFrame = script.Parent.Parent.HumanoidRootPart.CFrame

	-- Show penis
	char.Penis.Erecte.Shaft.Transparency = 0
	char.Penis.Erecte.Tip.Transparency = 1
	char.Penis.Sack.Union.Transparency = 0
	char.Penis.Sack.Union2.Transparency = 0
	char.Penis.Sack.Skin1.Transparency = 0

	script.Parent.Parent.Head.Face.Mouth.Transparency = 1

	-- Handle pants
	if char:FindFirstChild("Pants") then
		char.Pants.Parent = game.ReplicatedStorage
	end
end

local function StartBlowjob(valAnim, playerAnim)
	valAnim:Play()
	playerAnim:Play()
	valAnim.Looped = true
	playerAnim.Looped = true

	valAnim:GetMarkerReachedSignal("CloseEye"):Connect(function()
		script.Parent.Parent.Head.Face.Eyes.Texture = "rbxassetid://2801585179"
	end)

	valAnim:GetMarkerReachedSignal("Plap"):Connect(function()
		PlayRandomSound("Mouth", script.Parent.Parent.Head)
	end)

	valAnim:GetMarkerReachedSignal("Gag"):Connect(function()
		PlayRandomSound("Gag", script.Parent.Parent.Head)
	end)

	valAnim:GetMarkerReachedSignal("OpenEye"):Connect(function()
		script.Parent.Parent.Head.Face.Eyes.Texture = "rbxassetid://2801594973"
	end)
end

local function StopBlowjob(valAnim, playerAnim)
	valAnim:Stop()
	playerAnim:Stop()
	valAnim.Looped = false
	playerAnim.Looped = false
end

local function StartFastBlowjob(valAnim, playerAnim)
	valAnim:Play()
	playerAnim:Play()
	valAnim:AdjustSpeed(2.2)
	playerAnim:AdjustSpeed(2.2)
	valAnim.Looped = true
	playerAnim.Looped = true

	valAnim:GetMarkerReachedSignal("Plap"):Connect(function()
		PlayRandomSound("Mouth", script.Parent.Parent.Head)
	end)

	valAnim:GetMarkerReachedSignal("Gag"):Connect(function()
		PlayRandomSound("Gag", script.Parent.Parent.Head)
	end)
end

local function CumSequence(valAnim, playerAnim)
	valAnim:Play()
	playerAnim:Play()

	valAnim:GetMarkerReachedSignal("Gag"):Connect(function()
		PlayRandomSound("Gag", script.Parent.Parent.Head)
	end)

	valAnim:GetMarkerReachedSignal("Nut"):Connect(function()
		PlayRandomSound("Throat", script.Parent.Parent.Torso)
	end)

	valAnim:GetMarkerReachedSignal("WhiteFade"):Connect(function()
		game.Workspace.Events.WhiteFadeIn:Fire()
		wait(1.8)
		WhiteFade = true
		valAnim:Stop()
		playerAnim:Stop()
	end)
end

local function FinishScene(valAnim, char)
	char.HumanoidRootPart.CFrame = script.Parent.TorsoFinishPos.CFrame
	player.PlayerGui.Cumenant.Enabled = false
	player.PlayerGui.Cumenant.Frame.RiseNew.Enabled = false
	valAnim:Play()

	-- Hide penis
	char.Penis.Erecte.Shaft.Transparency = 1
	char.Penis.Erecte.Tip.Transparency = 1
	char.Penis.Sack.Union.Transparency = 1
	char.Penis.Sack.Union2.Transparency = 1
	char.Penis.Sack.Skin1.Transparency = 1

	game.Workspace.Events.ChangeCam:FireAllClients("Humanoid")
	script.Parent.Parent.Head.Face.Eyes.Texture = "rbxassetid://2801594973"
	script.Parent.Parent.Head.Face.Mouth.Transparency = 0

	if game.ReplicatedStorage:FindFirstChild("Pants") then
		game.ReplicatedStorage.Pants.Parent = char
	end

	wait(2)
	ChatService:Chat(script.Parent.Parent.Head, "Did you like that?")
	wait(2)
	valAnim:Stop()
	wait(2)
	script.Parent.Parent.HumanoidRootPart.Anchored = false
	ChatService:Chat(script.Parent.Parent.Head, "You need to talk to the gas station employee for that, come back soon~")
	game.Workspace.Events.ObjectiveUpdate:Fire("Go to the store")
	char.HumanoidRootPart.Anchored = false
	char.Humanoid.WalkSpeed = 16
	char.Humanoid.AutoRotate = true
end

-- Event: Touch to start dialogue
script.Parent.Touched:Connect(function(hit)
	if Debounce then return end

	player = game:GetService("Players"):GetPlayerFromCharacter(hit.Parent)
	if not player then return end

	Debounce = true
	game.Workspace.Events.ObjectiveUpdate:Fire("")
	hit.Parent.Humanoid.WalkSpeed = 0
	hit.Parent.Humanoid.AutoRotate = false
	player.PlayerGui.ValerieGui.Enabled = true
	player.PlayerGui.ValerieGui.Main.Enabled = true
	game.Lighting.ChatBlur.Enabled = true
end)

-- Event: Valerie sex
game.Workspace.Events.ValerieSex.Event:Connect(function()
	local char = player.Character
	local val = script.Parent.Parent

	-- Load animations
	local ValAnims = game.ReplicatedStorage.Anims:WaitForChild("ValerieAnims")
	local PlayerAnims = game.ReplicatedStorage.Anims:WaitForChild("PlayerAnims")

	local ValAnimBlow = ValAnims:WaitForChild("ValerieBlowJob"):Clone()
	local ValAnimBlowFinished = ValAnims:WaitForChild("ValerieBlowJobFinish"):Clone()
	local ValAnimBlowJobFastStart = ValAnims:WaitForChild("ValerieBlowJobFastStart"):Clone()
	local ValAnimBlowJobFast = ValAnims:WaitForChild("ValerieBlowJobFast"):Clone()
	local ValAnimBlowJobFastFinish = ValAnims:WaitForChild("ValerieBlowJobFastFinish"):Clone()
	local ValAnimRest = ValAnims:WaitForChild("ValerieRestPosititon"):Clone()

	local PlayerAnimBlowJob = PlayerAnims:WaitForChild("PlayerBlowJob"):Clone()
	local PlayerAnimBlowJobFinish = PlayerAnims:WaitForChild("PlayerBlowJobFinish"):Clone()
	local PlayerAnimBlowJobFastStart = PlayerAnims:WaitForChild("PlayerBlowJobFastStart"):Clone()
	local PlayerAnimBlowJobFast = PlayerAnims:WaitForChild("PlayerBlowJobFast"):Clone()
	local PlayerAnimBlowJobFastFinish = PlayerAnims:WaitForChild("PlayerBlowJobFastFinish"):Clone()

	ValAnimBlow.Parent = val.Anims
	ValAnimBlowFinished.Parent = val.Anims
	ValAnimBlowJobFastStart.Parent = val.Anims
	ValAnimBlowJobFast.Parent = val.Anims
	ValAnimBlowJobFastFinish.Parent = val.Anims
	ValAnimRest.Parent = val.Anims

	PlayerAnimBlowJob.Parent = char.Anims
	PlayerAnimBlowJobFinish.Parent = char.Anims
	PlayerAnimBlowJobFastStart.Parent = char.Anims
	PlayerAnimBlowJobFast.Parent = char.Anims
	PlayerAnimBlowJobFastFinish.Parent = char.Anims

	local ValAnimBlowPlay = val.Humanoid.Animator:LoadAnimation(ValAnimBlow)
	local ValAnimBlowFinishedPlay = val.Humanoid.Animator:LoadAnimation(ValAnimBlowFinished)
	local ValAnimBlowJobFastStartPlay = val.Humanoid.Animator:LoadAnimation(ValAnimBlowJobFastStart)
	local ValAnimBlowJobFastPlay = val.Humanoid.Animator:LoadAnimation(ValAnimBlowJobFast)
	local ValAnimBlowJobFastFinishPlay = val.Humanoid.Animator:LoadAnimation(ValAnimBlowJobFastFinish)
	local ValAnimRestPlay = val.Humanoid.Animator:LoadAnimation(ValAnimRest)

	local PlayerAnimBlowJobPlay = char.Humanoid.Animator:LoadAnimation(PlayerAnimBlowJob)
	local PlayerAnimBlowJobFinishPlay = char.Humanoid.Animator:LoadAnimation(PlayerAnimBlowJobFinish)
	local PlayerAnimBlowJobFastStartPlay = char.Humanoid.Animator:LoadAnimation(PlayerAnimBlowJobFastStart)
	local PlayerAnimBlowJobFastPlay = char.Humanoid.Animator:LoadAnimation(PlayerAnimBlowJobFast)
	local PlayerAnimBlowJobFastFinishPlay = char.Humanoid.Animator:LoadAnimation(PlayerAnimBlowJobFastFinish)

	-- Setup happens here, right before sex starts
	SetupCharacter(char)

	game.Workspace.Events.ChangeCam:FireAllClients("Head")
	game.Workspace.Events.CurrentPerson.Value = "Valerie"

	player.PlayerGui.Cumenant.Enabled = true
	player.PlayerGui.Cumenant.Frame.RiseNew.Enabled = true

	local Animation = true
	WhiteFade = false

	StartBlowjob(ValAnimBlowPlay, PlayerAnimBlowJobPlay)

	-- Event handlers
	game.Workspace.Events.ValerieSexFast.OnServerEvent:Connect(function()
		StopBlowjob(ValAnimBlowPlay, PlayerAnimBlowJobPlay)
		ValAnimBlowJobFastStartPlay:Play()
		PlayerAnimBlowJobFastStartPlay:Play()
		StartFastBlowjob(ValAnimBlowJobFastPlay, PlayerAnimBlowJobFastPlay)
	end)

	game.Workspace.Events.ValerieCum.OnServerEvent:Connect(function()
		Animation = false
		StopBlowjob(ValAnimBlowPlay, PlayerAnimBlowJobPlay)
		CumSequence(ValAnimBlowFinishedPlay, PlayerAnimBlowJobFinishPlay)
	end)

	game.Workspace.Events.ValerieCumFast.OnServerEvent:Connect(function()
		Animation = false
		StopBlowjob(ValAnimBlowJobFastPlay, PlayerAnimBlowJobFastPlay)
		CumSequence(ValAnimBlowJobFastFinishPlay, PlayerAnimBlowJobFastFinishPlay)
	end)

	repeat wait() until Animation == false
	repeat wait() until WhiteFade == true

	wait(3)
	game.Workspace.Events.WhiteFadeOut:Fire()
	FinishScene(ValAnimRestPlay, char)
end)
